CREATE TABLE Usuario(
    Id INT, 
    Nome VARCHAR(200),
    Email VARCHAR(100),
    Senha VARCHAR(100),
    PRIMARY KEY(Id)
);

CREATE TABLE Cliente(
    CodUsuario INT, 
    CPF CHAR(11),
    PRIMARY KEY(CPF),
    FOREIGN KEY(CodUsuario) REFERENCES Usuario(Id)
);

CREATE TABLE ProdutosAcompanhados(
    ClienteCPF CHAR(11),
    codProduto INT,
    FOREIGN KEY(ClienteCPF) REFERENCES Cliente(CPF)
);

CREATE TABLE Cartoes(
    ClienteCPF CHAR(11),
    numero CHAR(20),
    dataVencimento DATE,
    tipo TINYINT,
    PRIMARY KEY(numero), 
    FOREIGN KEY(ClienteCPF) REFERENCES Cliente(CPF)
);

CREATE TABLE Funcionario(
    CodFunc INT, 
    CodUsuario INT,
    PRIMARY KEY(CodFunc),
    FOREIGN KEY(CodUsuario) REFERENCES Usuario(Id)
);

CREATE TABLE Produto(
    Id INT,
    Valor MONEY,
    Descricao VARCHAR(300),
    Disponibilidade BIT,
    PRIMARY KEY(Id),
);

ALTER TABLE ProdutosAcompanhados
ADD FOREIGN(codProduto) REFERENCES Produto(Id);

CREATE TABLE Bebida(
    codBebida INT, 
    codProduto INT,
    Tipo VARCHAR(20),
    PRIMARY KEY(codBebida),
    FOREIGN KEY(codProduto) REFERENCES Produto(Id)
);

CREATE TABLE Tapioca(
    codTapioca INT, 
    codProduto INT, 
    Recheio VARCHAR(20),
    PRIMARY KEY(codTapioca),
    FOREIGN KEY(codProduto) REFERENCES Produto(Id)
);

CREATE TABLE Compra(
    Id INT, 
    Data DATE, 
    Status VARCHAR(50), 
    Valor MONEY, 
    CPFCliente CHAR(11), 
    CodFunc INT,
    PRIMARY KEY(Id),
    FOREIGN KEY(CPFCliente) REFERENCES Cliente(CPF),
    FOREIGN KEY(CodFunc) REFERENCES Funcionario(CodFunc)
);

CREATE TABLE ProdutosCompra(
    Id INT, 
    codCompra INT, 
    codProduto INT, 
    PRIMARY KEY (Id),
    FOREIGN KEY(codCompra) REFERENCES Compra(Id),
    FOREIGN KEY(codProduto) REFERENCES Produto(Id)
);
